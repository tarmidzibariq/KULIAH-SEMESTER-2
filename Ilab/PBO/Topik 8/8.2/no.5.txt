public class Pet {
    private String nama;
    private String hewan;
    private int umur;
    
    public void setNama(String nama) {
        this.nama = nama;
    }
    
    public void setHewan(String hewan) {
        this.hewan = hewan;
    }
    
    public void setUmur(int umur) {
        this.umur = umur;
    }
    
    public String getNama() {
        return nama;
    }
    
    public String getHewan() {
        return hewan;
    }
    
    public int getUmur() {
        return umur;
    }
}
